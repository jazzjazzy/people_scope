  39.     // {{{ class vars
  40.  
  41.     /**

  59.     // {{{ Mail_Queue_Container_mdb2()
  60.  
  61.     /**
  62.      * Contructor
  63.      *
  64.      * Mail_Queue_Container_mMDB2:: Mail_Queue_Container_mdb2()
  65.      *
  66.      * @param mixed $options    An associative array of connection option.
  67.      *
  68.      * @access public
  69.      */
  70.     function Mail_Queue_Container_mdb2($options)
  71.     {
  72.         if (!is_array($options) || !isset($options['dsn'])) {
  73.             return new Mail_Queue_Error(MAILQUEUE_ERROR_NO_OPTIONS,
  74.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  75.                 'No dns specified!');
  76.         }
  77.         if (isset($options['mail_table'])) {
  78.             $this->mail_table = $options['mail_table'];
  79.         }
  80.         $this->sequence = (isset($options['sequence']) ? $options['sequence'] : $this->mail_table);
  81.  
  82.         if (!empty($options['pearErrorMode'])) {
  83.             $this->pearErrorMode = $options['pearErrorMode'];
  84.         }
  85.         $this->db =& MDB2::connect($options['dsn'], true);
  86.         if (MDB2::isError($this->db)) {
  87.             return new Mail_Queue_Error(MAILQUEUE_ERROR_CANNOT_CONNECT,
  88.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  89.                 'MDB2::connect failed: '. MDB2::errorMessage($this->db));
  90.         } else {
  91.             $this->db->setFetchMode(MDB2_FETCHMODE_ASSOC);
  92.         }
  93.         $this->setOption();
  94.     }
  95.  
  96.     // }}}
  97.     // {{{ _preload()
  98.  
  99.     /**
  100.      * Preload mail to queue.
  101.      *
  102.      * @return mixed  True on success else Mail_Queue_Error object.
  103.      * @access private
  104.      */
  105.     function _preload()
  106.     {
  107.         $query = 'SELECT * FROM ' . $this->mail_table
  108.                 .' WHERE sent_time IS NULL AND try_sent < '. $this->try
  109.                 .' AND time_to_send < '.$this->db->quote(date("Y-m-d H:i:s"), 'timestamp')
  110.                 .' ORDER BY time_to_send';
  111.         $this->db->setLimit($this->limit, $this->offset);
  112.         $res = $this->db->query($query);
  113.         if (MDB2::isError($res)) {
  114.             return new Mail_Queue_Error(MAILQUEUE_ERROR_QUERY_FAILED,
  115.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  116.                 'MDB2::query failed - "'.$query.'" - '.MDB2::errorMessage($res));
  117.         }
  118.  
  119.         $this->_last_item = 0;
  120.         $this->queue_data = array(); //reset buffer
  121.         while ($row = $res->fetchRow(MDB2_FETCHMODE_ASSOC)) {
  122.             //var_dump($row['headers']);
  123.             if (!is_array($row)) {
  124.                 return new Mail_Queue_Error(MAILQUEUE_ERROR_QUERY_FAILED,
  125.                     $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  126.                     'MDB2::query failed - "'.$query.'" - '.MDB2::errorMessage($res));
  127.             }
  128.             $this->queue_data[$this->_last_item] = new Mail_Queue_Body(
  129.                 $row['id'],
  130.                 $row['create_time'],
  131.                 $row['time_to_send'],
  132.                 $row['sent_time'],
  133.                 $row['id_user'],
  134.                 $row['ip'],
  135.                 $row['sender'],
  136.                 $row['recipient'],
  137.                 unserialize($row['headers']),
  138.                 unserialize($row['body']),
  139.                 $row['delete_after_send'],
  140.                 $row['try_sent']
  141.             );
  142.             $this->_last_item++;
  143.         }
  144.  
  145.         return true;
  146.     }

  151.     /**
  152.      * Put new mail in queue and save in database.
  153.      *
  154.      * Mail_Queue_Container::put()
  155.      *
  156.      * @param string $time_to_send  When mail have to be send
  157.      * @param integer $id_user  Sender id
  158.      * @param string $ip  Sender ip
  159.      * @param string $from  Sender e-mail
  160.      * @param string $to  Reciepient e-mail
  161.      * @param string $hdrs  Mail headers (in RFC)
  162.      * @param string $body  Mail body (in RFC)
  163.      * @param bool $delete_after_send  Delete or not mail from db after send
  164.      *
  165.      * @return mixed  ID of the record where this mail has been put
  166.      *                 or Mail_Queue_Error on error
  167.      * @access public
  168.      ***/
  169.     function put($time_to_send, $id_user, $ip, $sender,
  170.                 $recipient, $headers, $body, $delete_after_send=true)
  171.     {
  172.         $id = $this->db->nextId($this->sequence);
  173.         if (empty($id) || MDB2::isError($id)) {
  174.             return new Mail_Queue_Error(MAILQUEUE_ERROR,
  175.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  176.                 'Cannot create id in: '.$this->sequence);
  177.         }
  178.         $query = 'INSERT INTO '. $this->mail_table
  179.                 .' (id, create_time, time_to_send, id_user, ip'
  180.                 .', sender, recipient, headers, body, delete_after_send) VALUES ('
  181.                 .       $this->db->quote($id, 'integer')
  182.                 .', ' . $this->db->quote(date("Y-m-d H:i:s"), 'timestamp')
  183.                 .', ' . $this->db->quote($time_to_send, 'timestamp')
  184.                 .', ' . $this->db->quote($id_user, 'integer')
  185.                 .', ' . $this->db->quote($ip, 'text')
  186.                 .', ' . $this->db->quote($sender, 'text')
  187.                 .', ' . $this->db->quote($recipient, 'text')
  188.                 .', ' . $this->db->quote($headers, 'text')   //clob
  189.                 .', ' . $this->db->quote($body, 'text')      //clob
  190.                 .', ' . ($delete_after_send ? 1 : 0)
  191.                 .')';
  192.         $res = $this->db->query($query);
  193.         if (MDB2::isError($res)) {
  194.             return new Mail_Queue_Error(MAILQUEUE_ERROR_QUERY_FAILED,
  195.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,
  196.                 'MDB2::query failed - "'.$query.'" - '.MDB2::errorMessage($res));
  197.         }
  198.         return $id;
  199.     }

  204.     /**

  211.     function countSend($mail)
  212.     {

  219.                 .' SET try_sent = ' . $this->db->quote($count, 'integer')

  221.         $res = $this->db->query($query);
  222.         if (MDB2::isError($res)) {
