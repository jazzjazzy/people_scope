  738.      * <pre>
  739.      * This method will first check the if the fields are in the valid_field array and strip out any that are not
  740.      * Then it check that fields that require a value in them from the fields_required have a value, if not add an error to validation_error array
  741.      * Then it will check all the values to find out if the value match the type found in the fields_validation_type array, if not add an error to validation_error array
  742.      * <div style="color:red">NOTE: This is generated information from the framework and will need to be corrected if there are any changes</div>
  743.      * </pre>
  744.      * 
  745.      * @see fields
  746.      * @see fields_required
  747.      * @see fields_validation_type
  748.      * @see validation_error
  749.      * 
  750.      * @param Array $request 
  751.      */
  752.     public function Validate($request){
  753.     
  754.         unset($this->valid_field);
  755.         unset($this->validation_error);
  756.         $isvalid = True;
  757.         
  758.         $validfields = $this->fields;
  759.         $requiredfields = $this->fields_required;
  760.         $fieldsvalidationtype = $this->fields_validation_type;
  761.         
  762.         foreach ($request AS $key=>$value){ //lets strip put unwanted or security violation fields  
  763.             if(in_array($key, $validfields)){
  764.                 $this->valid_field[$key] = $value; //pure fields
  765.             }
  766.         }
  767.         
  768.         foreach ($validfields AS $value){ //now lets just add fields as blank if they didn't come though so we can check them, helps with checkboxs
  769.             if(!isset($this->valid_field[$value])){
  770.                 $this->valid_field[$value] = ''; 
  771.             }
  772.         }
  773.         
  774.         if(count($requiredfields) > 0 ){
  775.             foreach($requiredfields AS $value){ // lets check all the required fields have a value 
  776.                 if (empty($this->valid_field[$value]) || $this->valid_field[$value] == 'NULL'){ 
  777.                     $this->validation_error[$value][] = 'Field is Required'; //error field
  778.                     $isvalid = false;
  779.                 }
  780.             }
  781.         }
  782.     
  783.         
  784.         
  785.         //now lets validate
  786.         foreach ($this->valid_field AS $key=>$value){
  787.             $value = trim($value);
  788.             if(!empty($value)){ // don't cheak if empty, alread done in required check 
  789.                 
  790.                 switch(@$fieldsvalidationtype[$key]){
  791.                     case 'TEXTAREA': if (strlen($value) > 1024) {
  792.                                     $this->validation_error[$key][] = 'Field longer then 1024 charactors'; 
  793.                                     $isvalid = false;
  794.                                 } break;
  795.                     case 'TEXT': if (strlen($value) > 1024) {
  796.                                     $this->validation_error[$key][] = 'Field longer then 1024 charactors'; 
  797.                                     $isvalid = false;
  798.                                 } break;
  799.                     case 'SAP': if ((!is_numeric($value)) || (strlen($value) != 10)) {
  800.                                     $this->validation_error[$key][] = 'not a valid SAP number'; 
  801.                                     $isvalid = false;
  802.                                 } break;
  803.                     case 'DECIMAL': if (!is_numeric($value)) {
  804.                                     $this->validation_error[$key][] = 'Decimal value expected';
