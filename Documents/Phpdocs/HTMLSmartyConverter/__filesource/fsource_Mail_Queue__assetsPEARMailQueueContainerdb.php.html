  116.         $query = sprintf("SELECT * FROM %s WHERE sent_time IS NULL
  117.                             AND try_sent < %d

  125.         if (DB::isError($query)) {

  127.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,

  130.         $res = $this->db->query($query);

  133.                 $this->pearErrorMode, E_USER_ERROR, __FILE__, __LINE__,

  149.                 $row['sent_time'],
  150.                 $row['id_user'],
